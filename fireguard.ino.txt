#include <WiFi.h>
#include <FirebaseESP32.h>
#include "DHT.h"

// 🔧 Pin configuration
#define DHTPIN 4
#define DHTTYPE DHT22
#define FLAME_DO 27   // Digital Output pin
#define FLAME_AO 34   // Analog Output pin
#define BUZZER 26
#define LED 25

// 🌐 Wi-Fi credentials
const char* ssid = "YOUR_WIFI_SSID";       // Replace with your Wi-Fi SSID
const char* password = "YOUR_WIFI_PASS";   // Replace with your Wi-Fi password

// 🔥 Firebase credentials
#define FIREBASE_HOST "your-project.firebaseio.com"   // Replace with your Firebase DB URL
#define FIREBASE_AUTH "your-secret-key"               // Replace with your Firebase secret key

FirebaseData fbData;
DHT dht(DHTPIN, DHTTYPE);

void connectWiFi() {
  Serial.print("Connecting to WiFi");
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\n✅ WiFi Connected!");
}

void setup() {
  Serial.begin(115200);
  pinMode(FLAME_DO, INPUT);
  pinMode(BUZZER, OUTPUT);
  pinMode(LED, OUTPUT);
  dht.begin();
  connectWiFi();
  Firebase.begin(FIREBASE_HOST, FIREBASE_AUTH);
  Firebase.reconnectWiFi(true);
}

void loop() {
  delay(2000); // allow sensor to stabilize
  float temp = dht.readTemperature();
  int flameDigital = digitalRead(FLAME_DO);
  int flameAnalog = analogRead(FLAME_AO);

  if (isnan(temp)) {
    Serial.println("⚠️ Sensor error!");
    return;
  }

  if (flameDigital == HIGH || temp > 50) {
    Serial.println("🔥 FIRE DETECTED!");
    digitalWrite(BUZZER, HIGH);
    digitalWrite(LED, HIGH);

    // Send alert to Firebase
    Firebase.setString(fbData, "/FireGuard/status", "FIRE DETECTED");
    Firebase.setFloat(fbData, "/FireGuard/temperature", temp);
    Firebase.setInt(fbData, "/FireGuard/flameAnalog", flameAnalog);
  } else {
    Serial.println("✅ SAFE");
    digitalWrite(BUZZER, LOW);
    digitalWrite(LED, LOW);

    Firebase.setString(fbData, "/FireGuard/status", "SAFE");
    Firebase.setFloat(fbData, "/FireGuard/temperature", temp);
    Firebase.setInt(fbData, "/FireGuard/flameAnalog", flameAnalog);
  }

  Serial.print("Temperature: ");
  Serial.println(temp);
  Serial.print("Flame Digital: ");
  Serial.println(flameDigital);
  Serial.print("Flame Analog: ");
  Serial.println(flameAnalog);
}
